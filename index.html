// Custom Error Classes export class ApiError extends Error { constructor(public type: ErrorType, public message: string = "error") { super(type); Error.captureStackTrace(this, this.constructor); } public static handle(err: ApiError, res: Response) { switch (err.type) { case ErrorType.BAD_REQUEST: return new BadRequestResponse(err.message).send(res); case ErrorType.UNAUTHORIZED: return new AuthFailureResponse(err.message).send(res); case ErrorType.NOT_FOUND: return new NotFoundResponse(err.message).send(res); default: return new InternalErrorResponse("Something went wrong").send(res); } } }
